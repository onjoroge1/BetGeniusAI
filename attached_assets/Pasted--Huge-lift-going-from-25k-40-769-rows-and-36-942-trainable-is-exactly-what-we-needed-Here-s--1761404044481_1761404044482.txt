A quick “gap report” to keep using:

WITH s AS (
  SELECT league_name, COUNT(*) AS matches,
         MIN(match_date) AS first_match, MAX(match_date) AS last_match
  FROM historical_odds GROUP BY league_name
)
SELECT league_name, matches, first_match, last_match,
       CASE WHEN last_match < DATE '2024-01-01' THEN 'STALE' ELSE 'OK' END freshness,
       CASE WHEN matches < 2000 THEN 'LOW_VOL' ELSE 'OK' END volume
FROM s
ORDER BY (freshness='STALE') DESC, (volume='LOW_VOL') DESC, matches ASC;

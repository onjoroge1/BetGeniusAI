Final Phase-2 QA (run now)

1) Closing coverage

-- Expect ≥1–3 samples per outcome in the last 5m window for KO’d matches
SELECT match_id, COUNT(*) AS samples
FROM clv_closing_feed
WHERE ts >= NOW() - INTERVAL '2 hours'
GROUP BY match_id
ORDER BY samples ASC
LIMIT 10;

2) VWAP vs fallback

-- Expect most settled rows to show LAST5_VWAP; fallback LAST_TICK only when samples < 3
SELECT closing_method, COUNT(*) 
FROM clv_realized
WHERE settled_at >= NOW() - INTERVAL '7 days'
GROUP BY closing_method;

3) Realized CLV sanity

-- Sanity: mean realized CLV should be ≥ 0 over time if alerts are genuine edges
SELECT AVG(realized_clv_pct) AS avg_realized, PERCENTILE_CONT(0.5) WITHIN GROUP (ORDER BY realized_clv_pct) AS p50
FROM clv_realized
WHERE settled_at >= NOW() - INTERVAL '7 days';

4) Books_used alignment

-- Check that alerts and closing samples both reflect independent desk groups
SELECT a.match_id, AVG(a.books_used) alert_books, AVG(f.books_used) close_books
FROM clv_alerts a
JOIN clv_closing_feed f USING (match_id)
WHERE a.created_at >= NOW() - INTERVAL '24 hours'
GROUP BY a.match_id
ORDER BY ABS(AVG(a.books_used)-AVG(f.books_used)) DESC
LIMIT 10;

# 3) FastAPI lifespan (modern, super fast startup)

If you want an even tidier pattern than `@on_event("startup")`, use lifespan. It binds the port first, then kicks off background stuff:

```python
# main.py
from contextlib import asynccontextmanager
from fastapi import FastAPI
import asyncio, logging, os

log = logging.getLogger("uvicorn.error")

def bg_enabled() -> bool:
    if os.getenv("REPLIT_DEPLOYMENT") == "1" and os.getenv("REPLIT_DEPLOYMENT_TYPE","").lower()=="autoscale":
        return os.getenv("ENABLE_BACKGROUND","0") == "1"
    return os.getenv("ENABLE_BACKGROUND","1") == "1"

@asynccontextmanager
async def lifespan(app: FastAPI):
    # startup — keep instant
    if not bg_enabled():
        log.info("🚫 AUTOSCALE MODE: background tasks DISABLED (API-only)")
        yield
        return

    async def _start_bg():
        # tiny delay lets port bind & health check pass
        await asyncio.sleep(1.0)
        # heavy imports happen only now
        from jobs.scheduler import start_scheduler
        log.info("⏳ Starting background scheduler after startup...")
        try:
            res = start_scheduler()
            if asyncio.iscoroutine(res):
                await res
            log.info("✅ Background scheduler started")
        except Exception:
            log.exception("Background scheduler failed")

    asyncio.create_task(_start_bg())
    yield  # server is live here
    # shutdown
    try:
        from jobs.scheduler import stop_scheduler
        stop_scheduler()
    except Exception:
        pass

app = FastAPI(lifespan=lifespan)

@app.get("/healthz")
async def healthz():
    return {"ok": True}
```

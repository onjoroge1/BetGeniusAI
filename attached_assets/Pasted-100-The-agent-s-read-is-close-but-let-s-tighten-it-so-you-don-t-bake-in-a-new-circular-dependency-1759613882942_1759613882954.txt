# Phase 3 — 3-step fixture resolver (breaks the cycle)

Use this order: **matches → snapshots → live APIF search**. Snapshots are **only a fallback**, never the primary.

```python
def resolve_apif_fixture_id(db, m):
    # 1) Canonical
    if m.api_football_fixture_id:
        return m.api_football_fixture_id

    # 2) Fallback from The Odds API snapshot (if your pipeline truly stores APIF id there)
    fid = db.fetch_val("""
        SELECT match_id::bigint
        FROM odds_snapshots
        WHERE match_internal_id = %(mid)s
          AND provider = 'the_odds_api'
        ORDER BY created_at DESC
        LIMIT 1
    """, {"mid": m.id})
    if fid:
        repo.update_fixture_id(m.id, fid)   # persist to matches
        return fid

    # 3) Live APIF search by league+date+teams (with aliases & normalization)
    fid = api_football.find_fixture_id_by_league_date_teams(
        league_id=m.league_id,
        kickoff_at=m.kickoff_at,
        home=m.home_team,
        away=m.away_team,
    )
    if fid:
        repo.update_fixture_id(m.id, fid)
    return fid
```

def collect_api_football_for_upcoming(db, horizon_hours=72):

1) Closing Line Value (CLV) — 1X2

What: Did our pick beat the close?
How (SQL sketch):

-- pick_time odds vs close odds, normalized to probs (vigless optional)
WITH tips AS (
  SELECT match_id, pick, book_id, odds_pick, tipped_at
  FROM model_tips WHERE market='1X2'
),
close AS (
  SELECT DISTINCT ON (match_id, ref_book) match_id, ref_book, 
         prob_h, prob_d, prob_a, closed_at
  FROM reference_close_1x2
  ORDER BY match_id, ref_book, closed_at DESC
)
SELECT t.match_id, t.pick,
       1/odds_pick                   AS prob_at_tip_raw,
       CASE t.pick WHEN 'H' THEN prob_h WHEN 'D' THEN prob_d ELSE prob_a END AS prob_at_close,
       (prob_at_close - 1/odds_pick) AS clv_prob_delta
FROM tips t JOIN close c USING (match_id);

Metric: CLV% = mean(prob_at_close - prob_at_tip_raw).
Good: >0.0% overall; >+1–2% on high-confidence buckets.

De-vig (1X2) – simple proportional method

def devig_1x2(pH, pD, pA):
    s = pH + pD + pA
    if s <= 0: return pH, pD, pA
    return pH/s, pD/s, pA/s

Reliability buckets

def reliability(df, bins=(0.35,0.45,0.55,0.65,0.75,0.85,0.95)):
    out=[]
    for lo,hi in zip(bins[:-1], bins[1:]):
        d=df[(df.max_prob>=lo)&(df.max_prob<hi)]
        if len(d)==0: continue
        out.append(dict(
            bin=f"{lo:.2f}-{hi:.2f}",
            n=len(d),
            avg_conf=d.max_prob.mean(),
            win_rate=d.accuracy.mean()
        ))
    return out

```python
def novig_from_decimal(odds):
    # odds: [oH, oD, oA]
    q = [1.0/max(1e-9,o) for o in odds]
    R = sum(q)
    return [x/R for x in q]
```

```python
def ev_and_clv(p_model, p_open, p_close, pick_idx):
    ev_close = p_model[pick_idx] - p_close[pick_idx]
    clv_prob = p_close[pick_idx] - p_open[pick_idx]
```

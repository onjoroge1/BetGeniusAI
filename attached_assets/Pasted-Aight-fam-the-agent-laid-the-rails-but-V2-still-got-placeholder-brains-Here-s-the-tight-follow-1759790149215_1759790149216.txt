5) Evaluation view: include model_version + CLV

Create/replace odds_accuracy_evaluation_v2:

CREATE OR REPLACE VIEW odds_accuracy_evaluation_v2 AS
WITH base AS (
  SELECT
    mil.match_id,
    m.league_id,
    mil.model_version,
    mil.p_home, mil.p_draw, mil.p_away,
    GREATEST(1e-9, mil.p_home) AS ph,  -- for logloss safety
    GREATEST(1e-9, mil.p_draw) AS pd,
    GREATEST(1e-9, mil.p_away) AS pa,
    r.home_goals, r.away_goals,
    CASE WHEN r.home_goals IS NULL THEN NULL
         WHEN r.home_goals > r.away_goals THEN 'H'
         WHEN r.home_goals = r.away_goals THEN 'D'
         ELSE 'A' END AS actual
  FROM model_inference_logs mil
  JOIN matches m ON m.match_id = mil.match_id
  LEFT JOIN match_results r ON r.match_id = mil.match_id
),
scored AS (
  SELECT *,
    CASE actual
      WHEN 'H' THEN -LN(ph)
      WHEN 'D' THEN -LN(pd)
      WHEN 'A' THEN -LN(pa)
      ELSE NULL END AS logloss,
    (
      POWER(ph - CASE WHEN actual='H' THEN 1 ELSE 0 END, 2) +
      POWER(pd - CASE WHEN actual='D' THEN 1 ELSE 0 END, 2) +
      POWER(pa - CASE WHEN actual='A' THEN 1 ELSE 0 END, 2)
    )/3.0 AS brier,
    CASE WHEN actual IS NOT NULL AND
              (ph>=pd AND ph>=pa AND actual='H' OR
               pd>=ph AND pd>=pa AND actual='D' OR
               pa>=ph AND pa>=pd AND actual='A')
         THEN 1 ELSE 0 END AS hit
  FROM base
),
clv AS (
  SELECT
    s.match_id,
    co.close_home_prob, co.close_draw_prob, co.close_away_prob
  FROM scored s
  LEFT JOIN closing_odds co ON co.match_id = s.match_id
)
SELECT
  s.*, 
  CASE WHEN clv.close_home_prob IS NULL THEN NULL
       WHEN s.ph > clv.close_home_prob AND (s.ph>=s.pd AND s.ph>=s.pa) THEN 1
       WHEN s.pd > clv.close_draw_prob AND (s.pd>=s.ph AND s.pd>=s.pa) THEN 1
       WHEN s.pa > clv.close_away_prob AND (s.pa>=s.ph AND s.pa>=s.pd) THEN 1
       ELSE 0 END AS clv_beats_close,
  CASE
    WHEN clv.close_home_prob IS NULL THEN NULL
    WHEN s.ph>=s.pd AND s.ph>=s.pa THEN s.ph - clv.close_home_prob
    WHEN s.pd>=s.ph AND s.pd>=s.pa THEN s.pd - clv.close_draw_prob
    ELSE s.pa - clv.close_away_prob
  END AS clv_edge
FROM scored s
LEFT JOIN clv ON clv.match_id = s.match_id;

(CLV: simple probability edge on the model’s top pick vs closing. Good enough for A/B gating.)

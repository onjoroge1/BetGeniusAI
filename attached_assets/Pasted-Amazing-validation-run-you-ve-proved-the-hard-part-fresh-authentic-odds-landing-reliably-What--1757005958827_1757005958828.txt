1) Fix consensus building (server-side rollup)

Add a small, idempotent consensus job that rolls odds_snapshots into consensus_predictions. This runs after every collection (and the 15-min filler).

Bucket helper (Postgres function):

CREATE OR REPLACE FUNCTION classify_bucket(secs_to_kickoff int)
RETURNS text AS $$
SELECT CASE
  WHEN $1 BETWEEN 7*24*3600 AND 8*24*3600 THEN '168h'
  WHEN $1 BETWEEN 5*24*3600 AND 6*24*3600 THEN '120h'
  WHEN $1 BETWEEN 66*3600   AND 78*3600   THEN '72h'
  WHEN $1 BETWEEN 40*3600   AND 56*3600   THEN '48h'
  WHEN $1 BETWEEN 18*3600   AND 30*3600   THEN '24h'
  WHEN $1 BETWEEN 10*3600   AND 14*3600   THEN '12h'
  WHEN $1 BETWEEN 4*3600    AND 8*3600    THEN '6h'
  WHEN $1 BETWEEN 2*3600    AND 4*3600    THEN '3h'
  WHEN $1 BETWEEN 0         AND 2*3600    THEN '1h'
  ELSE NULL END;
$$ LANGUAGE SQL IMMUTABLE;

Rollup (devig only complete triplets per book & bucket):

WITH snap AS (
  SELECT match_id, league_id, book_id, ts_snapshot,
         classify_bucket(secs_to_kickoff) AS time_bucket,
         outcome, odds_decimal
  FROM odds_snapshots
  WHERE market = 'h2h' AND period = 'FT' AND price_type = 'pre'
    AND ts_snapshot > NOW() - INTERVAL '48 hours'
),
latest_per_book AS (  -- one snapshot per book per bucket (closest to kickoff)
  SELECT DISTINCT ON (match_id, book_id, time_bucket)
         match_id, league_id, book_id, time_bucket, ts_snapshot
  FROM snap
  WHERE time_bucket IS NOT NULL
  ORDER BY match_id, book_id, time_bucket, ts_snapshot DESC
),
triplets AS (
  SELECT l.match_id, l.league_id, l.book_id, l.time_bucket,
         MAX(CASE WHEN s.outcome='H' THEN 1.0/s.odds_decimal END) AS p_h_raw,
         MAX(CASE WHEN s.outcome='D' THEN 1.0/s.odds_decimal END) AS p_d_raw,
         MAX(CASE WHEN s.outcome='A' THEN 1.0/s.odds_decimal END) AS p_a_raw
  FROM latest_per_book l
  JOIN snap s USING (match_id, book_id, time_bucket, ts_snapshot)
  GROUP BY 1,2,3,4
  HAVING COUNT(DISTINCT s.outcome) = 3
),
devig AS (
  SELECT match_id, league_id, time_bucket, book_id,
         p_h_raw/(p_h_raw+p_d_raw+p_a_raw) AS p_h,
         p_d_raw/(p_h_raw+p_d_raw+p_a_raw) AS p_d,
         p_a_raw/(p_h_raw+p_d_raw+p_a_raw) AS p_a
  FROM triplets
),
weighted AS (  -- optional: join book weights; else equal-average
  SELECT d.match_id, d.league_id, d.time_bucket,
         AVG(d.p_h) AS consensus_h,
         AVG(d.p_d) AS consensus_d,
         AVG(d.p_a) AS consensus_a,
         STDDEV_POP(d.p_h) AS dispersion_h,
         STDDEV_POP(d.p_d) AS dispersion_d,
         STDDEV_POP(d.p_a) AS dispersion_a,
         COUNT(*) AS n_books
  FROM devig d
  GROUP BY 1,2,3
)
INSERT INTO consensus_predictions
(match_id, time_bucket, consensus_h, consensus_d, consensus_a,
 dispersion_h, dispersion_d, dispersion_a, n_books, consensus_method, created_at)
SELECT match_id, time_bucket, consensus_h, consensus_d, consensus_a,
       COALESCE(dispersion_h,0), COALESCE(dispersion_d,0), COALESCE(dispersion_a,0),
       n_books, 'equal_mean', NOW()
FROM weighted
ON CONFLICT (match_id, time_bucket) DO UPDATE
SET consensus_h = EXCLUDED.consensus_h,
    consensus_d = EXCLUDED.consensus_d,
    consensus_a = EXCLUDED.consensus_a,
    dispersion_h = EXCLUDED.dispersion_h,
    dispersion_d = EXCLUDED.dispersion_d,
    dispersion_a = EXCLUDED.dispersion_a,
    n_books     = EXCLUDED.n_books,
    consensus_method = EXCLUDED.consensus_method,
    created_at  = NOW();

Scheduler: run this rollup right after each successful collect/fill job (and nightly backfill).

1) Closing sampler (every 30–60s, T-6m … T+2m)

Collect composite odds exactly like the producer (de-juiced, deduped, trimmed), but don’t emit alerts—just store one sample.

async def closing_sampler(now_utc):
    # pick fixtures in a small window around KO, e.g. now ∈ [KO-6m, KO+2m]
    fixtures = db.fetch_fixtures_in_window(now_utc, pre_minutes=6, post_minutes=2)

    for fx in fixtures:
        # build independent, fresh book probs (de-juiced), trimmed consensus
        composite = build_composite_probs(fx)  # {"H":pH,"D":pD,"A":pA}, books_used
        if not composite: 
            continue
        # convert to decimal odds and store 3 rows (H/D/A)
        odds = {k: 1.0/max(1e-12, v) for k, v in composite.items()}
        for k in ("H","D","A"):
            db.upsert_closing_sample(
                match_id=fx.match_id,
                ts=now_utc,
                outcome=k,
                composite_odds_dec=round(odds[k], 4),
                volume=None,                              # fill if you have it
                books_used=composite.meta.books_used,
                desk_groups=composite.meta.books_used
            )

2) Closing settle (T+2m or on results ingest)

Compute the closing odds from the last 5 minutes of samples.

CLOSE_WIN_SEC = 300

def compute_closing_odds(samples):  # list of (odds_dec, volume, ts)
    if len(samples) >= 3:
        # LAST5_VWAP (volume-weighted); fall back to time-weighted mean if volume is None
        if any(s.volume for s in samples):
            num = sum(s.odds_dec * (s.volume or 1.0) for s in samples)
            den = sum((s.volume or 1.0) for s in samples)
            return round(num/den, 4), "LAST5_VWAP", len(samples)
        else:
            # robust central tendency in case of spikes
            arr = sorted(s.odds_dec for s in samples)
            a = max(0, int(0.1*len(arr))); b = len(arr) - a
            trimmed = arr[a:b] if b > a else arr
            return round(sum(trimmed)/len(trimmed), 4), "LAST5_VWAP", len(samples)
    # Fallback
    last = max(samples, key=lambda s: s.ts) if samples else None
    return (last.odds_dec if last else None, "LAST_TICK", 1 if last else 0)

async def closing_settle(match_id, ko_utc):
    window_start = ko_utc - timedelta(seconds=CLOSE_WIN_SEC)
    window_end   = ko_utc + timedelta(seconds=60)      # grace 1m
    realized = {}
    for k in ("H","D","A"):
        samples = db.load_closing_samples(match_id, k, window_start, window_end)
        val, method, n = compute_closing_odds(samples)
        realized[k] = {"odds": val, "method": method, "n": n}
    return realized

Write clv_realized for each alert once per match:

def settle_alerts_for_match(match_id, realized, match_outcome):
    alerts = db.load_alerts_for_match(match_id)
    for a in alerts:
        close = realized[a.outcome]["odds"]
        if close is None: 
            continue
        realized_clv_pct = round(100.0 * (close - a.best_odds_dec) / a.best_odds_dec, 3)
        db.upsert_clv_realized(
            alert_id=a.alert_id,
            closing_odds_dec=close,
            realized_clv_pct=realized_clv_pct,
            closing_method=realized[a.outcome]["method"],
            closing_samples=realized[a.outcome]["n"],
            closing_window_sec=CLOSE_WIN_SEC,
            match_outcome=match_outcome,
            win=(match_outcome == a.outcome)
        )

@app.get("/clv/club/stats")
def clv_stats(league: str, window: str = "24h"):
    ts_from = now_utc() - parse_window(window)
    rows = db.q("""
      SELECT COUNT(*)::int AS n,
             AVG(books_used)::float AS books_used_avg,
             AVG(stability)::float AS stability_avg,
             AVG(clv_pct)::float AS clv_pct_avg
      FROM clv_alerts
      WHERE created_at >= :ts_from
        AND expires_at >= :now         -- still active or just emitted
        AND league = :league
    """, {"ts_from": ts_from, "now": now_utc(), "league": league})

    top = db.q("""
      SELECT match_id, outcome, clv_pct, books_used, stability, window_tag
      FROM clv_alerts
      WHERE created_at >= :ts_from AND league = :league
      ORDER BY clv_pct DESC LIMIT 10
    """, {"ts_from": ts_from, "league": league})

    return {
      "league": league,
      "window": window,
      "alerts_emitted": rows.n,
      "books_used_avg": round(rows.books_used_avg or 0, 2),
      "stability_avg":  round(rows.stability_avg or 0, 2),
      "clv_pct_avg":    round(rows.clv_pct_avg or 0, 2),
      "top_opportunities": [
        {"match_id": r.match_id, "outcome": r.outcome,
         "clv_pct": float(r.clv_pct), "books_used": r.books_used,
         "stability": float(r.stability), "window": r.window_tag}
        for r in top
      ]
    }

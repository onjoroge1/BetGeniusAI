# 2) Make the scheduler truly idempotent (no double runners)

In `scheduler.py`, ensure a single instance per process and guard against duplicate starts:

```py
# scheduler.py
import threading
import logging
logger = logging.getLogger(__name__)

_scheduler_lock = threading.Lock()
_scheduler_instance = None

def _start_jobs(sched):
    """
    Register ALL repeating jobs here, e.g.:
      - CLV alert producer (60s)
      - CLV closing sampler (60s)
      - CLV closing settler (60s)
      - Daily brief (00:05 UTC)
      - (Optional) real-time odds collectors
    """
    # IMPORTANT: check if already scheduled to avoid duplicates on hot-reload
    if getattr(sched, "_jobs_registered", False):
        return
    # 🔁 Every 60s jobs
    sched.every(60).seconds.do(run_clv_alert_producer)
    sched.every(60).seconds.do(run_clv_closing_sampler)
    sched.every(60).seconds.do(run_clv_closing_settler)
    # 🗓️ Daily brief
    sched.every().day.at("00:05").do(run_clv_daily_brief)
    sched._jobs_registered = True
    logger.info("Scheduler jobs registered.")

def get_background_scheduler():
    global _scheduler_instance
    with _scheduler_lock:
        if _scheduler_instance:
            return _scheduler_instance
        # create your scheduler (APScheduler, schedule, custom loop, etc.)
        sched = YourSchedulerImpl()
        _start_jobs(sched)
        sched.start()
        _scheduler_instance = sched
        logger.info("Background scheduler started.")
        return _scheduler_instance

def shutdown_background_scheduler():
    global _scheduler_instance
    with _scheduler_lock:
        if _scheduler_instance:
            _scheduler_instance.stop()
            _scheduler_instance = None
            logger.info("Background scheduler stopped.")
```

> If you run **multiple Uvicorn/Gunicorn workers**, only one should run the background scheduler. Use one of:
>
> * **SINGLETON via env**: only start when `os.getenv("RUN_SCHEDULER","1")=="1"` and set it for exactly one replica.
> * **DB advisory lock** (Postgres): try `pg_try_advisory_lock(42)` at startup; only start if lock acquired; release on shutdown.

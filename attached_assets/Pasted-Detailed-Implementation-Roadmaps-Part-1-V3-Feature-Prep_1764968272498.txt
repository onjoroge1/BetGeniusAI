Phase 1: Sharp Book Intelligence (Week 1-2)

**Goal**: Track Pinnacle odds separately and compute sharp/soft book divergence

1.1 Database Schema

```
-- Sharp book tracking table
CREATE TABLE sharp_book_odds (
    id SERIAL PRIMARY KEY,
    match_id INTEGER NOT NULL,
    sport VARCHAR(20) DEFAULT 'soccer',
    bookmaker VARCHAR(50) NOT NULL,  -- 'pinnacle', 'betfair', 'sbobet'
    is_sharp BOOLEAN DEFAULT true,

    -- Moneyline odds (decimal)
    odds_home NUMERIC(6,3),
    odds_draw NUMERIC(6,3),
    odds_away NUMERIC(6,3),

    -- Implied probabilities
    prob_home NUMERIC(5,4),
    prob_draw NUMERIC(5,4),
    prob_away NUMERIC(5,4),

    -- Market efficiency indicators
    overround NUMERIC(5,4),        -- Pinnacle typically 2-3%
    margin NUMERIC(5,4),

    ts_recorded TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    ts_kickoff TIMESTAMP WITH TIME ZONE,
    hours_before_kickoff NUMERIC(6,2),

    UNIQUE(match_id, bookmaker, ts_recorded)
);
CREATE INDEX idx_sharp_book_match ON sharp_book_odds(match_id);
CREATE INDEX idx_sharp_book_time ON sharp_book_odds(ts_recorded);

```

```
CREATE TABLE league_calibration (
    id SERIAL PRIMARY KEY,
    league_id INTEGER NOT NULL,
    league_name VARCHAR(100),

    ece_score NUMERIC(5,4),           -- Expected Calibration Error

    log_loss NUMERIC(6,4),            -- Log loss

    tier_weight NUMERIC(3,2),         -- 1.0, 0.8, 0.6, 0.4

    updated_at TIMESTAMP DEFAULT NOW(),

    UNIQUE(league_id, window_end)
);
CREATE INDEX idx_league_cal_league ON league_calibration(league_id);

```

Phase 3: Player & Injury Context (Week 3-4)

**Goal**: Capture team strength changes from injuries/suspensions

3.1 Database Schema

```
-- Injury tracking table
CREATE TABLE player_injuries (
    id SERIAL PRIMARY KEY,
    player_id INTEGER,
    player_name VARCHAR(100) NOT NULL,
    team_id INTEGER,
    team_name VARCHAR(100),
    league_id INTEGER,

    -- Injury details
    injury_type VARCHAR(50),         -- 'Injury', 'Suspension', 'International Duty'
    injury_reason VARCHAR(200),      -- 'Hamstring', 'Red Card', etc.

    -- Impact assessment
    player_value_rating NUMERIC(3,1), -- 1-10 importance score
    expected_return DATE,

    -- Match context
    fixture_id INTEGER,
    fixture_date DATE,

    -- Metadata
    source VARCHAR(50) DEFAULT 'api-football',
    ts_recorded TIMESTAMP DEFAULT NOW(),

    UNIQUE(player_id, fixture_id)
);
-- Team injury summary (pre-computed)
CREATE TABLE team_injury_summary (
    id SERIAL PRIMARY KEY,
    match_id INTEGER NOT NULL,
    team_id INTEGER NOT NULL,
    team_type VARCHAR(4),            -- 'home' or 'away'

    -- Aggregated metrics
    n_injured INTEGER DEFAULT 0,
    n_suspended INTEGER DEFAULT 0,
    total_impact_score NUMERIC(5,2), -- Sum of player ratings
    key_players_out TEXT[],          -- Array of key missing players

    ts_computed TIMESTAMP DEFAULT NOW(),

    UNIQUE(match_id, team_id)
);
CREATE INDEX idx_injuries_team ON player_injuries(team_id);
CREATE INDEX idx_injuries_fixture ON player_injuries(fixture_id);

```

```
    -- Totals (over/under)
    total_line NUMERIC(5,2),
    over_odds NUMERIC(6,3),
    under_odds NUMERIC(6,3),

    -- Bookmaker info
    bookmaker VARCHAR(50),
    n_bookmakers INTEGER,

    -- Metadata
    source VARCHAR(50) DEFAULT 'theoddsapi',
    ts_recorded TIMESTAMP DEFAULT NOW(),

    UNIQUE(event_id, bookmaker, ts_recorded)
);
CREATE INDEX idx_multisport_odds_sport ON multisport_odds_snapshots(sport);

CREATE INDEX idx_multisport_odds_time ON multisport_odds_snapshots(commence_time);
-- Multi-sport training data
CREATE TABLE multisport_training (
    id SERIAL PRIMARY KEY,
    sport VARCHAR(20) NOT NULL,
    sport_key VARCHAR(50) NOT NULL,
    event_id VARCHAR(100) NOT NULL,

    -- Match info
    home_team VARCHAR(100),
    away_team VARCHAR(100),
    match_date DATE,

    -- Result (2-way)
    home_score INTEGER,
    away_score INTEGER,
    outcome CHAR(1),  -- 'H' or 'A' (no draws in NBA/NHL regular time)

    -- Pre-computed features (JSONB for flexibility)
    features JSONB,

    created_at TIMESTAMP DEFAULT NOW(),

    UNIQUE(sport, event_id)
);
CREATE INDEX idx_multisport_training_sport ON multisport_training(sport);
```

```
"""
Multi-sport odds collector for NBA and NHL
Uses The Odds API - same quota as soccer
"""
class MultiSportCollector:
    """
    Collects odds for non-soccer sports

    Supported:
    - basketball_nba: NBA games
    - icehockey_nhl: NHL games
    - basketball_euroleague: European basketball (future)
    """

    SPORT_CONFIGS = {
        'basketball_nba': {
            'name': 'NBA',
            'markets': ['h2h', 'spreads', 'totals'],
            'regions': 'us',
            'outcome_type': '2way',  # No draws
            'typical_games_per_day': 8,
            'season': 'Oct-Jun'
        },
        'icehockey_nhl': {
            'name': 'NHL',
            'markets': ['h2h', 'spreads', 'totals'],
            'regions': 'us',
            'outcome_type': '2way',  # Moneyline is regulation only
            'typical_games_per_day': 10,
            'season': 'Oct-Jun'
        }
    }

    def __init__(self):
        self.api_key = os.getenv('ODDS_API_KEY')
        self.base_url = "https://api.the-odds-api.com/v4"
```

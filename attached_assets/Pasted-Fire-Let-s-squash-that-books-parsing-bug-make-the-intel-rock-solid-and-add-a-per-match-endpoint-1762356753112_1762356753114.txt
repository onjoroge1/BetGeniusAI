# 1) Fix the odds/`books` extraction (root cause of `'str' object has no attribute 'get'`)

Your `books` array is sometimes:

* a dict (`{"bookmaker":"bet365","prices":{...}}`)
* a stringified JSON blob
* or already a normalized probs object

Make the extractor resilient to all three.

```python
# utils/odds_extract.py
from typing import Dict, Tuple, Optional, Any, List
import json

OutcomeProbs = Dict[str, float]
OutcomeOdds  = Dict[str, float]

PREFERRED_KEYS = [
    # (path to decimal odds dict), supports multiple provider shapes
    ("prices",),                    # {"prices":{"home":2.05,"draw":3.40,"away":3.70}}
    ("odds", "decimal"),            # {"odds":{"decimal":{"home":2.05,...}}}
    ("odds",),                      # {"odds":{"home":2.05,"draw":3.40,"away":3.70}}
]

def _to_dict(maybe: Any) -> Optional[dict]:
    if isinstance(maybe, dict):
        return maybe
    if isinstance(maybe, str):
        maybe = maybe.strip()
        if maybe and (maybe[0] in "[{]" and maybe[-1] in "]}"):
            try:
                return json.loads(maybe)
            except Exception:
                return None
    return None

def _normalize_from_decimal_odds(o: OutcomeOdds) -> OutcomeProbs:
    inv = {k: (1.0 / v) for k, v in o.items() if v and v > 0}
    s = sum(inv.values()) or 1.0
    return {k: inv.get(k, 0.0) / s for k in ("home", "draw", "away")}

def extract_odds_and_probs(books: Any) -> Tuple[Optional[OutcomeOdds], Optional[OutcomeProbs], Optional[dict]]:
    """
    Returns: (decimal_odds, market_probs, best_book_obj)
    - decimal_odds: last usable decimal odds found
    - market_probs: normalized probabilities from decimal_odds
    - best_book_obj: the original book dict we used (or None)
    """
    # Normalize books -> list of dicts
    dict_books: List[dict] = []
    if isinstance(books, list):
        for b in books:
            d = _to_dict(b)
            if d:
                dict_books.append(d)
    else:
        d = _to_dict(books)
        if d:
            # sometimes a single object, wrap as list
            dict_books = [d]

    # Try each book in order (latest first preferred—feel free to reverse)
    for book in dict_books:
        # Already probabilities?
        # e.g., {"novig_current":{"home":0.45,"draw":0.28,"away":0.27}}
        for prob_key in ("novig_current", "probabilities", "market_probs"):
            mp = book.get(prob_key)
            if isinstance(mp, dict) and all(k in mp for k in ("home", "draw", "away")):
                return None, {k: float(mp[k]) for k in ("home", "draw", "away")}, book

        # Otherwise, try to find decimal odds with known paths
        for path in PREFERRED_KEYS:
            node = book
            ok = True
            for key in path:
                if isinstance(node, dict) and key in node:
                    node = node[key]
                else:
                    ok = False
                    break
            if ok and isinstance(node, dict) and all(k in node for k in ("home", "draw", "away")):
                dec = {k: float(node[k]) for k in ("home", "draw", "away")}
                return dec, _normalize_from_decimal_odds(dec), book

        # Fallback: a flat object with decimal odds?
        if all(k in book for k in ("home", "draw", "away")):
            try:
                dec = {k: float(book[k]) for k in ("home", "draw", "away")}
                return dec, _normalize_from_decimal_odds(dec), book
            except Exception:
                pass

    # Nothing usable
    return None, None, None
```

# 2) Make it work “per match id”

Add a single-match endpoint and wire the same compute. You’ve got two nice paths:

### A) Add a new endpoint

```python
# routes/betting_intelligence.py
from fastapi import APIRouter, Query, HTTPException
from utils.odds_extract import extract_odds_and_probs
from utils.betting_edge import compute_betting_intel

router = APIRouter(prefix="/betting-intelligence", tags=["betting-intelligence"])

@router.get("/{match_id}")
def betting_intel_match(
    match_id: int,
    model: str = Query("best"),
    bankroll: float | None = Query(None),
    kelly: float = Query(0.5)  # fractional kelly
):
    match = db.get_match_with_predictions_and_books(match_id)  # your DAO
    if not match:
        raise HTTPException(status_code=404, detail="Match not found")

    # choose model probs (v2_selective else v1_consensus)
    model_probs = pick_model_probs(match, model=model)  # implement pick logic

    decimal_odds, market_probs, used_book = extract_odds_and_probs(match.get("books"))
    if not (model_probs and (market_probs or decimal_odds)):
        return {"match_id": match_id, "betting_intelligence": None}

    bi = compute_betting_intel(
        model_probs=model_probs,
        market_probs=market_probs,
        decimal_odds=decimal_odds,
        bankroll=bankroll,
        kelly_frac=kelly,
    )
    return {
        "match_id": match_id,
        "home": match["home"],
        "away": match["away"],
        "league": match.get("league"),
        "kickoff_time": match.get("kickoff_time"),
        "model_used": model if model != "best" else bi["best_bet"]["pick"],  # or return actual model name
        "betting_intelligence": bi,
        "best_odds_book": used_book and used_book.get("bookmaker"),
    }
```

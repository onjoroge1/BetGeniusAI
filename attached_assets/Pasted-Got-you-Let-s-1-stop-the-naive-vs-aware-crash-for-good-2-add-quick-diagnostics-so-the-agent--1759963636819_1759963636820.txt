Symptom

Root
Your API-Football integration compares/uses datetimes with mixed tz awareness during fixture upsert.

Hard fix (one place, used everywhere):

# utils/dates.py
from datetime import datetime, timezone
from typing import Optional, Union

def as_aware_utc(dt: Optional[Union[datetime, str, int, float]]) -> Optional[datetime]:
    """
    Normalize input to tz-aware UTC datetime.
    Accepts: datetime (naive or aware), ISO string, or epoch seconds.
    """
    if dt is None:
        return None
    if isinstance(dt, (int, float)):
        return datetime.fromtimestamp(dt, tz=timezone.utc)
    if isinstance(dt, str):
        # API-Football returns ISO-like; rely on fromisoformat if possible
        try:
            d = datetime.fromisoformat(dt.replace("Z", "+00:00"))
        except ValueError:
            # last resort: strip micros/spaces
            d = datetime.strptime(dt.split(".")[0], "%Y-%m-%dT%H:%M:%S")
        return d.astimezone(timezone.utc) if d.tzinfo else d.replace(tzinfo=timezone.utc)
    if isinstance(dt, datetime):
        return dt if dt.tzinfo else dt.replace(tzinfo=timezone.utc)
    raise TypeError(f"Unsupported datetime type: {type(dt)}")

B) “Why did clv_alerts disappear?” — targeted diagnostics

These quick queries tell us whether alerts were expired, deleted, or cascaded.

-- 1) Did a job truncate or DDL-reset the table?
SELECT to_char(pg_stat_all_tables.last_vacuum, 'YYYY-MM-DD HH24:MI:SS') AS last_vacuum,
       to_char(pg_stat_all_tables.last_autovacuum, 'YYYY-MM-DD HH24:MI:SS') AS last_autovacuum,
       to_char(pg_stat_all_tables.last_analyze, 'YYYY-MM-DD HH24:MI:SS') AS last_analyze
FROM pg_stat_all_tables
WHERE relname = 'clv_alerts';

-- 2) Were they simply expired by TTL?
SELECT COUNT(*) AS active,
       COUNT(*) FILTER (WHERE expires_at <= NOW()) AS expired
FROM clv_alerts;

-- 3) Was there a schema recreate?
SELECT to_char(pg_class.relrewrite, 'YYYY-MM-DD HH24:MI:SS') AS rewrote_at
FROM pg_class WHERE relname='clv_alerts';

-- 4) Are we cascading deletes from parent entities?
SELECT conname, confrelid::regclass AS parent_table
FROM pg_constraint
WHERE conrelid='clv_alerts'::regclass AND contype='f';

-- 5) Did our app delete (audit recent deletions)?
-- Only works if you have an audit table; otherwise add this trigger for future:

(Optional) Add a tiny audit trigger for future debugging

CREATE TABLE IF NOT EXISTS clv_alerts_audit(
  id bigserial, alert_id bigint, action text, at timestamptz default now()
);

CREATE OR REPLACE FUNCTION audit_clv_alerts_del()
RETURNS trigger LANGUAGE plpgsql AS $$
BEGIN
  INSERT INTO clv_alerts_audit(alert_id, action) VALUES (OLD.id, 'DELETE');
  RETURN OLD;
END $$;

DO $$
BEGIN
  IF NOT EXISTS (
    SELECT 1 FROM pg_trigger WHERE tgname = 'trg_audit_clv_alerts_del'
  ) THEN
    CREATE TRIGGER trg_audit_clv_alerts_del
    BEFORE DELETE ON clv_alerts
    FOR EACH ROW EXECUTE FUNCTION audit_clv_alerts_del();
  END IF;
END $$;

Common causes we’ve seen:
	•	TTL too short: expires_at set to created_at + 5min → table looks “wiped”.
	•	A nightly backfill script re-creates tables/views.
	•	A cascade from fixtures or matches cleanups.

If it’s TTL: increase via env (e.g., 15–30 min) and re-run producer.

# tests/test_closing_loop.py
import asyncpg, os, asyncio

async def _run():
    conn = await asyncpg.connect(os.getenv("DATABASE_URL"))
    # precondition: there is at least one fixture within T-6..T+2
    m = await conn.fetchrow("""
      SELECT match_id FROM fixtures
      WHERE kickoff_at BETWEEN NOW()-INTERVAL '2 minutes' AND NOW()+INTERVAL '6 minutes'
      LIMIT 1
    """)
    if not m:
      await conn.close()
      return "SKIPPED"
    mid = m["match_id"]
    # sampler should write
    s = await conn.fetchrow("SELECT COUNT(*) c FROM clv_closing_feed WHERE match_id=$1 AND ts>NOW()-INTERVAL '10 minutes'", mid)
    # aggregator should upsert closing_odds
    co = await conn.fetchrow("SELECT COUNT(*) c FROM closing_odds WHERE match_id=$1", mid)
    await conn.close()
    return (s["c"], co["c"])

def test_closing_loop():
    res = asyncio.run(_run())
    assert res == "SKIPPED" or (res[0] >= 1 and res[1] >= 1)

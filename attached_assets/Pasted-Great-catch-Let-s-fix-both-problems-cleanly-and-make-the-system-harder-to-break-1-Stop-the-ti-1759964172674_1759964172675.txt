B. Normalize datetimes at ingest (API-Football)

Use a single helper (you can drop this in utils/dates.py):

from datetime import datetime, timezone
def as_aware_utc(dt):
    if dt is None: return None
    if isinstance(dt, (int, float)):
        return datetime.fromtimestamp(dt, tz=timezone.utc)
    if isinstance(dt, str):
        d = datetime.fromisoformat(dt.replace("Z","+00:00"))
        return d if d.tzinfo else d.replace(tzinfo=timezone.utc)
    if isinstance(dt, datetime):
        return dt if dt.tzinfo else dt.replace(tzinfo=timezone.utc)
    raise TypeError(type(dt))

Then call it everywhere you read or compare times (fixture upsert, odds writer, producer filters):

B. Replace hard deletes with soft archive (no more “table looks wiped”)

Create a history table + move expired rows there. This makes debugging easy and keeps the live table lean.

CREATE TABLE IF NOT EXISTS clv_alerts_history (LIKE clv_alerts INCLUDING ALL);

-- Upsert-friendly unique key if you have one (optional)
-- ALTER TABLE clv_alerts_history ADD CONSTRAINT clv_alerts_history_pk PRIMARY KEY (id);

Cleanup job SQL (run every 5–10 min):

WITH moved AS (
  INSERT INTO clv_alerts_history
  SELECT * FROM clv_alerts
  WHERE expires_at <= NOW() - INTERVAL '10 minutes'  -- grace buffer
  RETURNING id
)
DELETE FROM clv_alerts a
USING moved m
WHERE a.id = m.id;

C. Rebuild the dedupe index for timestamptz

If you dedupe by (match_id, outcome, minute bucket), recreate the object on timestamptz.

-- Drop old one (name may differ)
DROP INDEX IF EXISTS idx_clv_alerts_dedupe;

-- Prefer a GENERATED column for stability & planner friendliness
ALTER TABLE clv_alerts
  ADD COLUMN IF NOT EXISTS created_minute timestamptz
  GENERATED ALWAYS AS (date_trunc('minute', created_at)) STORED;

CREATE UNIQUE INDEX IF NOT EXISTS idx_clv_alerts_dedupe
ON clv_alerts (match_id, outcome, created_minute);

If your schema doesn’t allow GENERATED columns, you can keep a functional index:

CREATE UNIQUE INDEX IF NOT EXISTS idx_clv_alerts_dedupe
ON clv_alerts (match_id, outcome, date_trunc('minute', created_at));



D. Add a partial index for “active” alerts (speeds API)

CREATE INDEX IF NOT EXISTS idx_clv_alerts_active
ON clv_alerts (expires_at DESC)
WHERE expires_at > NOW();

async def archive_and_delete_expired_alerts(pool):

def test_alert_ttl_and_archive(pg_conn):
    r = pg_conn.fetchrow("""
      INSERT INTO clv_alerts(match_id,outcome,clv_bps,stability,created_at,expires_at,best_book_id,snapshot_price,composite_price)
      VALUES (999001,'H',150,0.95, NOW(), NOW()+INTERVAL '1 minute', 1, 2.10, 2.30)
      RETURNING id;
    """); alert_id = r["id"]
    # simulate cleaner run after expiry
    pg_conn.execute("""
      WITH moved AS (
        INSERT INTO clv_alerts_history SELECT * FROM clv_alerts
        WHERE id=$1 AND expires_at <= NOW() + INTERVAL '0 minute'
        RETURNING id
      )
      DELETE FROM clv_alerts a USING moved m WHERE a.id=m.id;
    """, alert_id)
    live = pg_conn.fetchval("SELECT COUNT(*) FROM clv_alerts WHERE id=$1", alert_id)
    hist = pg_conn.fetchval("SELECT COUNT(*) FROM clv_alerts_history WHERE id=$1", alert_id)
    assert live == 0 and hist == 1

2) Fix probability construction (common failure point)

Make sure you’re converting decimal odds → implied probs → margin-stripped probs:

def probs_from_decimal(o_home, o_draw, o_away):
    # guard against zeros/None
    if min(o_home, o_draw, o_away) is None or min(o_home, o_draw, o_away) <= 1.0:
        return None
    inv = [1.0/o_home, 1.0/o_draw, 1.0/o_away]
    overround = sum(inv)
    p = [x/overround for x in inv]          # **margin-stripped**
    # sanity: sum≈1, each in (0,1)
    if not (0.99 <= sum(p) <= 1.01) or min(p) <= 0 or max(p) >= 0.99:
        return None
    return p  # [p_home, p_draw, p_away]

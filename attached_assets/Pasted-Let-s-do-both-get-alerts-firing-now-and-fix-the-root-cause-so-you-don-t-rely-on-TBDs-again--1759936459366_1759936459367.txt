### Option B (derive from odds payload)

If books include team strings in the market/outcome payload, take the **majority label** from the last 5–10 minutes.

```sql
-- one-shot SQL used by Phase B (or a helper):
WITH names AS (
  SELECT match_id,
         MAX(home_name) FILTER (WHERE home_name IS NOT NULL) AS home_guess,
         MAX(away_name) FILTER (WHERE away_name IS NOT NULL) AS away_guess
  FROM (
    SELECT os.match_id,
           CASE WHEN os.outcome='H' THEN os.team_label END AS home_name,
           CASE WHEN os.outcome='A' THEN os.team_label END AS away_name
    FROM odds_snapshots os
    WHERE os.ts_snapshot > NOW() - INTERVAL '10 minutes'
  ) z
  GROUP BY match_id
)
UPDATE fixtures f
SET home_team = COALESCE(NULLIF(f.home_team,'TBD'), n.home_guess),
    away_team = COALESCE(NULLIF(f.away_team,'TBD'), n.away_guess)
FROM names n
WHERE f.match_id = n.match_id
  AND (f.home_team ILIKE 'TBD%' OR f.away_team ILIKE 'TBD%');
```

## 3) Backfill today’s TBDs (one-shot)

Run this once so current candidates render nicely:

```sql
-- Backfill from recent odds-derived labels (safe if team_label exists)
WITH latest AS (
  SELECT DISTINCT ON (match_id, outcome)
         match_id, outcome, team_label
  FROM odds_snapshots
  WHERE ts_snapshot > NOW() - INTERVAL '12 hours'
  ORDER BY match_id, outcome, ts_snapshot DESC
),
pivot AS (
  SELECT
    match_id,
    MAX(team_label) FILTER (WHERE outcome='H') AS home_guess,
    MAX(team_label) FILTER (WHERE outcome='A') AS away_guess
  FROM latest GROUP BY match_id
)
UPDATE fixtures f
SET home_team = COALESCE(NULLIF(f.home_team,'TBD'), p.home_guess),
    away_team = COALESCE(NULLIF(f.away_team,'TBD'), p.away_guess)
FROM pivot p
WHERE f.match_id = p.match_id
  AND (f.home_team ILIKE 'TBD%' OR f.away_team ILIKE 'TBD%');
```

(If `team_label` doesn’t exist, use the API-Football enrich function once for those 10 match_ids.)

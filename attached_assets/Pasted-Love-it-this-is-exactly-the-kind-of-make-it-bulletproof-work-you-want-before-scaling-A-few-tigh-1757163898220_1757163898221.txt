	2.	Backfill by “missing buckets”, not “missing matches”
Your agent noted “no candidates” because the finder looked only for matches with zero consensus. Instead, target bucket gaps per match:

-- Candidates where a match has snapshots in a bucket window but no consensus for that bucket
WITH snaps AS (
  SELECT
    match_id,
    CASE
      WHEN secs_to_kickoff BETWEEN 60*60*60 AND 84*60*60 THEN '72h'
      WHEN secs_to_kickoff BETWEEN 36*60*60 AND 60*60*60 THEN '48h'
      WHEN secs_to_kickoff BETWEEN 18*60*60 AND 30*60*60 THEN '24h'
      WHEN secs_to_kickoff BETWEEN 10*60*60 AND 14*60*60 THEN '12h'
      WHEN secs_to_kickoff BETWEEN  5*60*60 AND  7*60*60 THEN '6h'
      WHEN secs_to_kickoff BETWEEN  2*60*60 AND  4*60*60 THEN '3h'
    END AS bucket,
    COUNT(*) FILTER (WHERE outcome IN ('H','D','A')) AS rows_all,
    COUNT(DISTINCT book_id) FILTER (
      WHERE outcome IN ('H','D','A')
    ) AS books_all
  FROM odds_snapshots
  WHERE ts_snapshot > NOW() - INTERVAL '90 days'
  GROUP BY 1,2
),
gaps AS (
  SELECT s.match_id, s.bucket, s.books_all
  FROM snaps s
  LEFT JOIN consensus_predictions c
    ON c.match_id = s.match_id AND c.time_bucket = s.bucket
  WHERE s.bucket IS NOT NULL
    AND s.books_all >= 2
    AND c.match_id IS NULL
)
SELECT * FROM gaps ORDER BY bucket, books_all DESC LIMIT 200;

Feed that list to your backfill job; expect consensus rows to jump fast.

WITH odds_m AS (
  SELECT DISTINCT match_id FROM odds_snapshots WHERE ts_snapshot > NOW() - INTERVAL '24 hours'
),
cons_m AS (
  SELECT DISTINCT match_id FROM consensus_predictions WHERE created_at > NOW() - INTERVAL '24 hours'

SELECT
  (SELECT COUNT(*) FROM odds_m) AS matches_with_odds_24h,
  (SELECT COUNT(*) FROM cons_m) AS matches_with_consensus_24h,

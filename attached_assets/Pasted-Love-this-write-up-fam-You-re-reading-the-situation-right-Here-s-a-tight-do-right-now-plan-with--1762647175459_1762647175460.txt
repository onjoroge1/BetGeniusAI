# 2) Add true opening vs latest (enable drift/vol)

Create two views (or materialized views) off your **real** snapshots.

```sql
-- Odds within [5m, 4h] pre-KO as "latest"
CREATE MATERIALIZED VIEW IF NOT EXISTS odds_real_latest AS
SELECT s.match_id,
       (SUM(p_home)/SUM(p_home+p_draw+p_away)) AS p_last_home,
       (SUM(p_draw)/SUM(p_home+p_draw+p_away)) AS p_last_draw,
       (SUM(p_away)/SUM(p_home+p_draw+p_away)) AS p_last_away,
       COUNT(DISTINCT s.bookmaker) AS n_books_last
FROM odds_snapshots s
JOIN fixtures f USING (match_id)
WHERE s.ts_effective <= f.kickoff_at
  AND s.ts_effective >= f.kickoff_at - INTERVAL '4 hours'
  AND s.ts_effective >= f.kickoff_at - INTERVAL '5 minutes' * 48 -- keep window flexible if needed
GROUP BY s.match_id;

-- Earliest pre-KO snapshot as "opening"
CREATE MATERIALIZED VIEW IF NOT EXISTS odds_real_opening AS
WITH first_snap AS (
  SELECT s.*,
         ROW_NUMBER() OVER (PARTITION BY s.match_id ORDER BY s.ts_effective ASC) AS rn
  FROM odds_snapshots s
  JOIN fixtures f USING (match_id)
  WHERE s.ts_effective < f.kickoff_at
)
SELECT match_id,
       (p_home/(p_home+p_draw+p_away)) AS p_open_home,
       (p_draw/(p_home+p_draw+p_away)) AS p_open_draw,
       (p_away/(p_home+p_draw+p_away)) AS p_open_away
FROM first_snap
WHERE rn = 1;
```

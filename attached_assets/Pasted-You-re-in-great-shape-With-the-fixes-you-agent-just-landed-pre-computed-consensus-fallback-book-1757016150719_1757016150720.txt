What to monitor daily (simple SQL you can paste)

Consensus coverage by bucket (24h):

SELECT time_bucket, COUNT(*) matches, ROUND(AVG(n_books),1) avg_books
FROM consensus_predictions
WHERE created_at > NOW() - INTERVAL '24 hours'
  AND market='h2h' AND period='FT'
GROUP BY time_bucket ORDER BY 1;

Sanity/degeneracy:

SELECT COUNT(*) bad
FROM consensus_predictions
WHERE market='h2h' AND period='FT'
  AND (consensus_h IS NULL OR consensus_d IS NULL OR consensus_a IS NULL
       OR consensus_h<0 OR consensus_d<0 OR consensus_a<0
       OR consensus_h>1 OR consensus_d>1 OR consensus_a>1);

Gaps (snapshots exist but no consensus):

WITH recent AS (
  SELECT DISTINCT match_id
  FROM odds_snapshots
  WHERE ts_snapshot > NOW() - INTERVAL '24 hours'
    AND market='h2h' AND period='FT'
)
SELECT r.match_id
FROM recent r
LEFT JOIN consensus_predictions c
  ON c.match_id=r.match_id
 AND c.created_at > NOW() - INTERVAL '24 hours'
WHERE c.match_id IS NULL
LIMIT 50;

🔧 Fix — make heavy jobs non-blocking & on a timer

1) Patch your scheduler to use background tasks (no await on heavy jobs)

# utils/scheduler.py  (or wherever your loop lives)
import asyncio, logging, os
from datetime import datetime, timezone, timedelta
logger = logging.getLogger(__name__)

PHASE_B_INTERVAL   = 60      # s
PRODUCER_INTERVAL  = 60
SAMPLER_INTERVAL   = 60
SETTLER_INTERVAL   = 60
CLEANUP_INTERVAL   = 300     # 5 min
PHASE_B_TIMEOUT    = 300     # 5 min hard cap

class EnhancedScheduler:
    def __init__(self):
        self.is_running = False
        self.tasks = {}             # name -> asyncio.Task
        self.last_run = {}          # name -> datetime
        self._lock = asyncio.Lock() # reentrancy guard

    async def _spawn(self, name: str, coro, timeout: int | None = None):
        """Spawn task once; skip if still running. Wrap with timeout + logging."""
        if task := self.tasks.get(name):
            if not task.done():
                return  # already running

        async def wrapper():
            started = datetime.now(timezone.utc)
            logger.info(f"▶️ {name}: start")
            try:
                if timeout:
                    await asyncio.wait_for(coro(), timeout=timeout)
                else:
                    await coro()
                self.last_run[name] = datetime.now(timezone.utc)
                logger.info(f"✅ {name}: ok in {(self.last_run[name]-started).total_seconds():.1f}s")
            except asyncio.TimeoutError:
                logger.error(f"⏱️ {name}: TIMEOUT after {timeout}s")
            except Exception as e:
                logger.exception(f"💥 {name}: failed: {e}")

        self.tasks[name] = asyncio.create_task(wrapper())

    async def _tick(self):
        now = datetime.now(timezone.utc)
        # Schedule Phase B (non-blocking)
        if self._due("phase_b", now, PHASE_B_INTERVAL):
            await self._spawn("phase_b", self._run_phase_b_fresh_odds, PHASE_B_TIMEOUT)

        # CLV producer
        if self._due("producer", now, PRODUCER_INTERVAL):
            await self._spawn("producer", self._run_clv_producer)

        # Closing sampler
        if self._due("sampler", now, SAMPLER_INTERVAL):
            await self._spawn("sampler", self._run_closing_sampler)

        # Settler / aggregator
        if self._due("settler", now, SETTLER_INTERVAL):
            await self._spawn("settler", self._run_closing_settler)

        # TTL cleanup (archive → delete)
        if self._due("cleanup", now, CLEANUP_INTERVAL):
            await self._spawn("cleanup", self._run_clv_ttl_cleanup)

    def _due(self, name: str, now: datetime, interval: int) -> bool:
        last = self.last_run.get(name)
        return (last is None) or ((now - last).total_seconds() >= interval)

    async def run_forever(self):
        if self.is_running: return
        self.is_running = True
        logger.info("🚀 Enhanced scheduler started")
        try:
            while self.is_running:
                await self._tick()
                await asyncio.sleep(1)  # fine-grained loop; tasks run in background
        finally:
            self.is_running = False
            logger.info("🛑 Scheduler stopped")

Key change: no await on _run_phase_b_fresh_odds() in the loop. We spawn it with asyncio.create_task() and a timeout, so producer/cleanup/sampler/settler all get time slices.

3) Cleanup job = archive first, then delete (you already built it, just wire it)

async def _run_clv_ttl_cleanup(self):
    # archive then delete (grace 10 min)
    sql = """
    WITH moved AS (
      INSERT INTO clv_alerts_history
      SELECT * FROM clv_alerts
      WHERE expires_at <= NOW() - INTERVAL '10 minutes'
      RETURNING id
    )
    DELETE FROM clv_alerts a
    USING moved m
    WHERE a.id = m.id;
    """
    await self.db.execute(sql)
    # log counts to make behavior obvious
    c_hist = await self.db.fetchval("SELECT COUNT(*) FROM clv_alerts_history WHERE created_at > NOW()-INTERVAL '1 hour'")
    c_live = await self.db.fetchval("SELECT COUNT(*) FROM clv_alerts WHERE expires_at > NOW()")
    logger.info(f"🧹 cleanup: history_1h={c_hist}, live_active={c_live}")

    async def _run_closing_sampler(self):     await asyncio.sleep(0.01); self.hit.append("sampler")
    async def _run_closing_settler(self):     await asyncio.sleep(0.01); self.hit.append("settler")
    async def _run_clv_ttl_cleanup(self):     await asyncio.sleep(0.01); self.hit.append("cleanup")

async def _run():
    s = Dummy()
    task = asyncio.create_task(s.run_forever())
    await asyncio.sleep(2.5)   # let several ticks happen
    s.is_running = False
    await task
    return s.hit

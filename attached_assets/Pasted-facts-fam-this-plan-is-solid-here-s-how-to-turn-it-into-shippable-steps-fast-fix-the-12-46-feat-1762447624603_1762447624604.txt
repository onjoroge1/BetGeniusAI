# features/v2_feature_builder.py
from functools import lru_cache
import pandas as pd
import sqlalchemy as sa

engine = sa.create_engine("postgresql+psycopg2://...")

@lru_cache(maxsize=2048)
def _elo_window(team_id: int, kickoff: str, w: int = 10) -> dict:
    q = """
    with hist as (
      select result_points, goals_for, goals_against
      from training_matches
      where team_id = :tid and kickoff_at < :ko
      order by kickoff_at desc
      limit :w
    )
    select
      avg(result_points)::float as form_pts,
      avg(goals_for)::float as g_for,
      avg(goals_against)::float as g_against
    from hist;
    """
    row = engine.execute(sa.text(q), {"tid": team_id, "ko": kickoff, "w": w}).mappings().first()
    return dict(row or {"form_pts": None, "g_for": None, "g_against": None})

def _rest_days(team_id: int, kickoff: str) -> float:
    q = """
      select extract(epoch from (:ko - kickoff_at))/86400.0 as rd
      from training_matches
      where team_id = :tid and kickoff_at < :ko
      order by kickoff_at desc limit 1;
    """
    row = engine.execute(sa.text(q), {"tid": team_id, "ko": kickoff}).scalar()
    return float(row) if row is not None else None

1B) EV/CLV evaluation harness (drop-in)

Create eval/eval_v2.py to score what matters (logloss, Brier, ECE, profit, CLV).

# eval/eval_v2.py
import numpy as np
import pandas as pd
from sklearn.metrics import log_loss, brier_score_loss

def expected_value(prob: float, price: float) -> float:
    # price is decimal odds; EV per $1 stake
    return prob * (price - 1) - (1 - prob) * 1

def ece(y_true_proba, y_pred_proba, bins=10):
    # multiclass ECE
    y = np.argmax(y_true_proba, axis=1)
    p = y_pred_proba
    conf = p.max(axis=1)
    pred = p.argmax(axis=1)
    acc = (pred == y).astype(float)
    bin_ids = np.floor(conf * bins).clip(0, bins-1).astype(int)
    ece = 0.0
    for b in range(bins):
        m = bin_ids == b
        if m.any():
            ece += abs(acc[m].mean() - conf[m].mean()) * (m.mean())
    return ece

def evaluate(df: pd.DataFrame):
    # df columns: y_true ∈ {0,1,2}, proba_0/1/2, price_0/1/2, closing_price_0/1/2
    y = df["y_true"].values
    Y = np.eye(3)[y]
    P = df[["proba_0","proba_1","proba_2"]].values
    ll = log_loss(y, P)
    bs = np.mean([brier_score_loss((y==k).astype(int), P[:,k]) for k in range(3)])
    cal = ece(Y, P)

    # naive betting policy: bet outcome with highest EV if EV > 0
    mx_idx = np.argmax(P * (df[["price_0","price_1","price_2"]].values - 1) - (1 - P), axis=1)
    evs = []
    clvs = []
    for i, k in enumerate(mx_idx):
        prob = P[i, k]; price = df.iloc[i, [df.columns.get_loc(f"price_{k}")]].item()
        close = df.iloc[i, [df.columns.get_loc(f"closing_price_{k}")]].item()
        evs.append(expected_value(prob, price))
        # CLV (in probability space): 1/close - 1/price OR simple: close - price
        clvs.append(close - price)
    return {"logloss": ll, "brier": bs, "ece": cal, "avg_ev": float(np.mean(evs)), "avg_clv_delta": float(np.mean(clvs))}

# calibration/calibrate.py
from sklearn.isotonic import IsotonicRegression
def segment_calibrate(y_true, proba, segment):
    models = {}
    calibrated = np.zeros_like(proba)
    for s in np.unique(segment):
        m = segment == s
        for k in range(proba.shape[1]):
            iso = IsotonicRegression(out_of_bounds='clip')
            iso.fit(proba[m, k], (y_true==k).astype(int)[m])
            calibrated[m, k] = iso.transform(proba[m, k])
        models[s] = "fitted"
    # renormalize row-wise
    calibrated = calibrated / calibrated.sum(axis=1, keepdims=True)
    return calibrated

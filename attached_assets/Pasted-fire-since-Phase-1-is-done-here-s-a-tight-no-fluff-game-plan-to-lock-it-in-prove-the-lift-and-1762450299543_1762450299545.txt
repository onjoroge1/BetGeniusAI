PHASE-2 — enrichment rails (ship schema now, fill it continuously)

A) “Gap discovery” SQL (paste into your backfill agent)

Missing lineups / availability

-- matches with no player_availability
select m.match_id, m.league_id, m.season, m.kickoff_at
from matches m
left join player_availability pa on pa.match_id = m.match_id
where m.kickoff_at < now()
group by m.match_id, m.league_id, m.season, m.kickoff_at
having count(pa.player_id) = 0
order by m.kickoff_at desc
limit 2000;

Missing referee

select m.match_id, m.league_id, m.season, m.kickoff_at
from matches m
left join referees r on r.ref_id = m.ref_id
where m.kickoff_at < now() and (m.ref_id is null)
order by m.kickoff_at desc
limit 2000;

Missing weather

select m.match_id, m.venue_lat, m.venue_lon, m.kickoff_at
from matches m
left join match_weather w on w.match_id = m.match_id
where m.kickoff_at < now() and w.match_id is null
order by m.kickoff_at desc
limit 2000;

Missing schedule context

select m.match_id, m.home_team_id, m.away_team_id, m.kickoff_at
from matches m
left join match_context c on c.match_id = m.match_id
where m.kickoff_at < now() and c.match_id is null
order by m.kickoff_at desc
limit 2000;

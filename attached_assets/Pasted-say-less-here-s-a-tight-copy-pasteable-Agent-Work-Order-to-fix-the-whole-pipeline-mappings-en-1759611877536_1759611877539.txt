### 3.1 league_map accessors

Create `models/league_map.py`:

```python
from typing import Optional, Any

def get_theodds_sport_key(db, league_id: int) -> Optional[str]:
    row = db.fetch_one("""
        SELECT theodds_sport_key
        FROM league_map
        WHERE (api_football_league_id = %(lid)s OR league_id = %(lid)s)
          AND active = TRUE
        LIMIT 1
    """, {"lid": league_id})
    return row["theodds_sport_key"] if row and row["theodds_sport_key"] else None

def get_api_football_league_id(db, league_id: int) -> Optional[int]:
    row = db.fetch_one("""
        SELECT api_football_league_id
        FROM league_map
        WHERE (api_football_league_id = %(lid)s OR league_id = %(lid)s)
          AND active = TRUE
        LIMIT 1
    """, {"lid": league_id})
    return row["api_football_league_id"] if row and row["api_football_league_id"] else None
```

### 3.2 Enrichment must run **before** collection

Update `trigger_manual_collection.py` (or the manual flow it calls):

```python
from models.league_map import get_theodds_sport_key
from services.api_football import enrich_fixture_ids_for_matches, collect_odds_for_fixture
from services.odds_api import collect_odds_for_match

def run_manual_multi_source_collection(db, horizon_hours: int = 72):
    matches = repo.get_upcoming_matches(horizon_hours=horizon_hours)  # status NS in window

    # 1) Enrich fixture_ids (writes back)
    to_enrich = [m for m in matches if not m.api_football_fixture_id]
    if to_enrich:
        enriched_count = enrich_fixture_ids_for_matches(db, to_enrich)  # returns int
        log.info(f"🔧 Enriched {enriched_count}/{len(to_enrich)} fixture IDs")

    # 2) Collect from The Odds API (using league_map resolver)
    odds_snapshots = 0
    for m in matches:
        sport_key = get_theodds_sport_key(db, m.league_id)
        if not sport_key:
            log.error(f"❌ CRITICAL: League {m.league_id} not mapped to Odds API sport key!")
            continue
        odds_snapshots += collect_odds_for_match(db, m, sport_key)

    # 3) Collect from API-Football (fixture_id based; with fallback)
    apif_rows = 0
    for m in matches:
        fid = m.api_football_fixture_id
        if not fid:
            # fallback: try soft resolve just-in-time
            fid = services.api_football.try_resolve_fixture_id(db, m)
            if fid:
                repo.update_fixture_id(m.id, fid)
                log.warning(f"⚠️ Late fixture_id resolve for match {m.id} → {fid}")
            else:
                log.warning(f"⚠️ No fixture_id for match {m.id} ({m.home_team} vs {m.away_team})")
                continue
        apif_rows += collect_odds_for_fixture(db, fixture_id=fid, kickoff_at=m.kickoff_at)

    log.info(f"✅ MULTI-SOURCE collection completed: TheOdds={odds_snapshots} | API-Football={apif_rows}")
    return odds_snapshots, apif_rows
```

### 3.3 API-Football enrichment service

Implement team aliasing + league/date guardrails in `services/api_football.py`:

```python
def enrich_fixture_ids_for_matches(db, matches) -> int:
    count = 0
    for m in matches:
        # Use league + kickoff date window + team aliases for precision
        lid = get_api_football_league_id(db, m.league_id) or m.league_id
        fixture_id = find_fixture_id_by_league_date_teams(
            league_id=lid,
            kickoff_at=m.kickoff_at,
            home=m.home_team,
            away=m.away_team,
        )
        if fixture_id:
            repo.update_fixture_id(m.id, fixture_id)
            count += 1
    return count

def try_resolve_fixture_id(db, m):
    lid = get_api_football_league_id(db, m.league_id) or m.league_id
    return find_fixture_id_by_league_date_teams(
        league_id=lid,
        kickoff_at=m.kickoff_at,
        home=m.home_team,
        away=m.away_team,
    )
```

> `find_fixture_id_by_league_date_teams()` should normalize accents, strip punctuation, and check an alias table (`team_aliases`) to map “LAFC” ↔ “Los Angeles FC”, “Pumas UNAM” ↔ “U.N.A.M. - Pumas”, etc. Use a ±2h start-time window to avoid mis-matches.

-- Upcoming by league with fixture_id coverage
SELECT m.league_id,
       COUNT(*)                            AS upcoming,
       SUM(CASE WHEN m.api_football_fixture_id IS NOT NULL THEN 1 ELSE 0 END) AS with_fixture_id
FROM matches m
WHERE m.status = 'NS'
  AND m.kickoff_at BETWEEN NOW() AND NOW() + INTERVAL '72 hours'
GROUP BY 1
ORDER BY 1;

-- Provider freshness (last 24h)
SELECT provider, COUNT(*) AS rows_24h
FROM odds_snapshots
WHERE created_at >= NOW() - INTERVAL '24 hours'
GROUP BY provider;

# 3) Python — adaptive staleness + time-boxed TBD + tiered min-books

Patch `clv_alert_producer.py` (names adjusted to your style):

```python
from datetime import datetime, timezone, timedelta
from typing import Dict

MIN_STALENESS = int(os.getenv("CLV_MIN_STALENESS_SEC", 600))
MAX_STALENESS = int(os.getenv("CLV_MAX_STALENESS_SEC", 7200))
TBD_ALLOW_BEFORE_HOURS = int(os.getenv("CLV_TBD_ALLOW_BEFORE_HOURS", 36))
ALERT_COOLDOWN_MIN = int(os.getenv("CLV_ALERT_COOLDOWN_MIN", 20))

LEAGUE_TIERS: Dict[str, str] = {
    **{k: v for k, v in (os.getenv("CLV_LEAGUE_TIERS_JSON") and json.loads(os.getenv("CLV_LEAGUE_TIERS_JSON")) or {}).items()}
}
LEAGUE_MIN_BOOKS: Dict[str, int] = {
    "tier1": 6, "tier2": 4, "tier3": 3,
    **(os.getenv("CLV_LEAGUE_MIN_BOOKS_JSON") and json.loads(os.getenv("CLV_LEAGUE_MIN_BOOKS_JSON")) or {})
}

def _seconds_to_ko(ko: datetime) -> int:
    now = datetime.now(timezone.utc)
    return max(0, int((ko - now).total_seconds()))

def _adaptive_staleness(ko: datetime) -> int:
    secs = _seconds_to_ko(ko)
    # 15% of time-to-KO, clamped to [MIN_STALENESS, MAX_STALENESS]
    raw = max(MIN_STALENESS, round(0.15 * secs))
    return max(MIN_STALENESS, min(raw, MAX_STALENESS))

def _allow_tbd_for_ko(ko: datetime) -> bool:
    return _seconds_to_ko(ko) > (TBD_ALLOW_BEFORE_HOURS * 3600)

def _tier_for_league(league_id: str) -> str:
    return LEAGUE_TIERS.get(str(league_id), "tier2")

def _min_books_for_league(league_id: str) -> int:
    return LEAGUE_MIN_BOOKS.get(_tier_for_league(league_id), 4)
```

# 4) Alert de-dupe — cooldown + window_tag

Before inserting a new alert, enforce cooldown. One simple technique is to bucket time into 20-minute windows and use the unique index:

```python
def _window_tag(now_utc: datetime, cooldown_min: int = ALERT_COOLDOWN_MIN) -> str:
    epoch_bucket = int(now_utc.timestamp() // (cooldown_min * 60))
    return f"{cooldown_min}m-{epoch_bucket}"

def create_alert_if_fresh(match_id, market, side, payload):
    now_utc = datetime.now(timezone.utc)
    window_tag = _window_tag(now_utc)

    try:
        db.execute(
            """
            INSERT INTO alerts (match_id, market, side, window_tag, payload, created_at, is_active)
            VALUES (%s, %s, %s, %s, %s::jsonb, NOW(), true)
            ON CONFLICT ON CONSTRAINT ux_alerts_key_window DO NOTHING;
            """,
            (match_id, market, side, window_tag, json.dumps(payload)),
        )
        return True
    except Exception as e:
        logger.warning("alert insert skipped (likely duplicate): %s", e)
        return False
```

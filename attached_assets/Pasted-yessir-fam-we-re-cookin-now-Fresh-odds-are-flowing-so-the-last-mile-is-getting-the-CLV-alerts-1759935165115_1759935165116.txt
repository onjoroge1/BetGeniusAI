# Debug: are there alert candidates right now?

Run these to see **why** 0 alerts:

**A) Do we have enough fresh books per match?**

```sql
WITH recent AS (
  SELECT match_id, book_id,
         MAX(ts_snapshot) AS ts
  FROM odds_snapshots
  WHERE ts_snapshot > NOW() - INTERVAL '10 minutes'
  GROUP BY match_id, book_id
)
SELECT r.match_id,
       COUNT(DISTINCT r.book_id) AS fresh_books,
       MIN(r.ts) AS first_fresh_ts,
       MAX(r.ts) AS last_fresh_ts
FROM recent r
JOIN fixtures f USING(match_id)
WHERE f.kickoff_at BETWEEN NOW() AND NOW() + INTERVAL '12 hours'
  AND f.home_team NOT ILIKE 'TBD%' AND f.away_team NOT ILIKE 'TBD%'
GROUP BY r.match_id
ORDER BY last_fresh_ts DESC
LIMIT 50;
```

**B) Spot a quick edge vs ‚Äúcomposite‚Äù (median) across books**
*(pre-closing proxy so producer has something to chew on)*

```sql
WITH latest AS (
  SELECT os.match_id, os.outcome, os.book_id,
         FIRST_VALUE(os.odds_decimal) OVER (
           PARTITION BY os.match_id, os.outcome, os.book_id
           ORDER BY os.ts_snapshot DESC
         ) AS price
  FROM odds_snapshots os
  WHERE os.ts_snapshot > NOW() - INTERVAL '10 minutes'
),
comp AS (
  SELECT match_id, outcome,
         PERCENTILE_CONT(0.5) WITHIN GROUP (ORDER BY price) AS composite_price
  FROM latest
  GROUP BY match_id, outcome
),
edges AS (
  SELECT l.match_id, l.book_id, l.outcome, l.price,
         c.composite_price,
         (1.0/c.composite_price - 1.0/l.price) AS clv_pct,
         10000*(1.0/c.composite_price - 1.0/l.price) AS clv_bps
  FROM latest l JOIN comp c USING (match_id, outcome)
)
SELECT e.match_id,
       MAX(CASE WHEN outcome='home' THEN clv_bps END) AS home_bps,
       MAX(CASE WHEN outcome='draw' THEN clv_bps END) AS draw_bps,
       MAX(CASE WHEN outcome='away' THEN clv_bps END) AS away_bps,
       COUNT(DISTINCT book_id) AS books
FROM edges e
GROUP BY e.match_id
HAVING COUNT(DISTINCT book_id) >= 3
ORDER BY GREATEST(COALESCE(home_bps,-1e9),COALESCE(draw_bps,-1e9),COALESCE(away_bps,-1e9)) DESC
LIMIT 25;
```

If this returns rows with **bps ‚â• your MIN_EDGE_BPS**, the producer should be creating alerts. If not, lower `CLV_MIN_EDGE_BPS` for a bit to validate the pipeline.
